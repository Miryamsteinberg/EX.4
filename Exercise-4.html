Exploring the count tables

Here, import an example count table and do some exploration of the expression data.

###```{r exSetup1, eval=FALSE} counts_file <- system.file(“extdata/rna-seq/SRP029880.raw_counts.tsv”, package = “compGenomRData”) coldata_file <- system.file(“extdata/rna-seq/SRP029880.colData.tsv”, package = “compGenomRData”) ###

  1. Normalize the counts using the TPM approach.
library(pheatmap)
## Warning: package 'pheatmap' was built under R version 4.2.2
library(stats)
library(corrplot)
## Warning: package 'corrplot' was built under R version 4.2.2
## corrplot 0.92 loaded
counts_file <- read.table("C:/Users/meirilab/Downloads/SRP029880.raw_counts.tsv")
coldata_file <- read.table("C:/Users/meirilab/Downloads/SRP029880.colData.tsv",sep = "\t" )

counts <- as.matrix(counts_file)

geneLengths <- as.vector(subset(counts, select = c(width)))
                         
#Computing TPM
#find gene length normalized values

rpk <- apply(subset(counts,select = c(-width)),2,
              function(x) x/(geneLengths/1000))

#normalize by the sample size using rpk values

tpm <- apply(rpk, 2, function(x) x / sum(as.numeric(x)) * 10^6)
  1. Plot a heatmap of the top 500 most variable genes. Compare with the heatmap obtained using the 100 most variable genes. Hint : use var() function to compute the variance.
Vec1 <- apply(tpm, 1, var)

selectedGenes500 <- names(Vec1[order(Vec1, decreasing = T)][1:500])

pheatmap(tpm[selectedGenes500,], scale = 'row', show_rownames = FALSE)

selectedGenes100 <- names(Vec1[order(Vec1, decreasing = T)][1:100])

pheatmap(tpm[selectedGenes100,], scale = 'row', show_rownames = FALSE)

  1. Re-do the heatmaps setting the scale argument to none, and column. Compare the results with scale = 'row'.
pheatmap(tpm[selectedGenes100,], scale = 'none', show_rownames = FALSE)

pheatmap(tpm[selectedGenes100,], scale = 'column', show_rownames = FALSE)

  1. Draw a correlation plot for the samples depicting the sample differences as ‘ellipses’, drawing only the upper end of the matrix, and order samples by hierarchical clustering results based on average linkage clustering method.
correlationMatrix <- cor(tpm)


corrplot(correlationMatrix, order = 'hclust', method = 'ellipse', hclust.method = 'average',type = 'upper',
         addrect = 2, addCoef.col = 'white',
         number.cex = 0.7)
